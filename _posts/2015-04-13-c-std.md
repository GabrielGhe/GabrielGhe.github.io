#include <vector>

std::vector<int> v;
v.push_back(3);   // [3]
v.push_back(6);   // [3,6]
v.pop_back();     // [3]
v.push_back(7);   // [3,7]
v.push_back(9);   // [3,7,9]

// Deletes the second element (g[1])
std::vector<int> g = {2,5,8,11,14};
g.erase(g.begin() + 1); // [2,8,11,14]

// Deletes the 2nd through 3rd elements (vec[1], vec[2])
g.erase(g.begin() + 1, g.begin() + 3);

// size
